// API'den konum bilgilerini almak için asenkron bir fonksiyon oluşturun async function konumAl() { try { // ipapi.co'nun sağladığı ücretsiz API ile IP üzerinden konum verisi alınır const response = await fetch('https://ipapi.co/json/'); // Yanıt başarılı değilse hata döndür if (!response.ok) throw new Error('Konum bilgisi alınamadı'); // Yanıt verilerini JSON formatında çözümle const data = await response.json(); // Konum bilgilerini konsola yazdır console.log('Ülke:', data.country_name); console.log('Bölge:', data.region); console.log('Şehir:', data.city); console.log('IP:', data.ip); } catch (error) { console.error('Hata:', error); } } // Fonksiyonu çalıştır konumAl();
